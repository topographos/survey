# 2550 to 2000
nb_2550 <- nearNeighb(sites_2550_BC$xcoord, sites_2550_BC$ycoord, sites_2550_BC$id, k = 4)

sites_2550_BC_DF = st_drop_geometry(sites_2550_BC)

sdom_2550BC  = lapply(nb_2550, function(x) {
  nb_2550 <- merge(x, sites_2550_BC_DF, by.x = "nb", by.y = "id")
  focal_size <- unique(sites_2550_BC_DF[sites_2550_BC_DF$id == nb_2550$focal,"category"])
  dbhDom(nb_2550$category,focal_size)
  
})

sdom_2550BC = as.data.frame(do.call(rbind, sdom_2550BC))

sdom_2550BC = sdom_2550BC %>% 
  transmute(
    id = rownames(sdom_2550BC),
    dom = V1
  )
